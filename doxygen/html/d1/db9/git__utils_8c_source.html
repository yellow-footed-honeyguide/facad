20/**
21 * @brief Checks if the current directory is part of a Git repository.
22 *
23 * This function executes the 'git rev-parse' command to determine
24 * if the current working directory is within a Git repository.
25 *
26 * @return true if the current directory is in a Git repository, false otherwise.
27 */
28bool is_git_repository() {
29 // Execute 'git rev-parse' command and redirect output to /dev/null
30 FILE* fp = popen("git rev-parse --is-inside-work-tree 2>/dev/null", "r");
31 if (fp == NULL) {
32 return false; // Return false if command execution failed
33 }
34
35 char buffer[128]; // Buffer to store command output
36 // Read command output and check if it's "true"
37 bool result = fgets(buffer, sizeof(buffer), fp) != NULL && strcmp(buffer, "true\n") == 0;
38 pclose(fp); // Close the command pipe
39 return result; // Return the result
40}

42/**
43 * @brief Parses a Git status string and returns a single character status code.
44 *
45 * This function interprets the two-character Git status string and
46 * returns a single character representing the file's status.
47 *
48 * @param status_str Two-character Git status string.
49 * @return Single character representing the file's Git status.
50 */
51static char parse_status(const char* status_str) {
52 if (status_str[0] == '?' && status_str[1] == '?') return 'U'; // Untracked
53 if (status_str[0] == '!' && status_str[1] == '!') return 'I'; // Ignored
54 return status_str[0] != ' ' ? status_str[0] : status_str[1]; // Modified, Staged, etc.
55}

57/**
58 * @brief Retrieves the Git status for all files in the current repository.
59 *
60 * This function executes 'git status' command and parses its output to create
61 * a list of files with their corresponding Git status.
62 *
63 * @return GitFileList structure containing the status of all files in the repository.
64 */
65GitFileList get_git_status() {
66 GitFileList list = {0}; // Initialize an empty GitFileList
67
68 // Execute 'git status' command
69 FILE* fp = popen("git status -s", "r");
70 if (fp == NULL) {
71 return list; // Return empty list if command execution failed
72 }
73
74 char line[MAX_FILENAME + 3]; // Buffer to store each line of output
75 // Read each line of the command output
76 while (fgets(line, sizeof(line), fp) != NULL && list.count < MAX_FILES) {
77 if (strlen(line) > 3) { // Ensure the line has enough characters
78 GitFile file;
79 file.status = parse_status(line); // Parse the status
80 // Copy the filename, skipping the first 3 characters (status and space)
81 strncpy(file.filename, line + 3, MAX_FILENAME - 1);
82 file.filename[strcspn(file.filename, "\n")] = 0; // Remove newline
83 list.files[list.count++] = file; // Add file to the list
84 }
85 }
86
87 pclose(fp); // Close the command pipe
88 return list; // Return the populated list
89}

91/**
92 * @brief Retrieves the name of the current Git branch.
93 *
94 * This function executes 'git rev-parse' command to get the name of the
95 * current Git branch.
96 *
97 * @return Dynamically allocated string containing the branch name, or NULL if an error occurred.
98 * @note The caller is responsible for freeing the returned string.
99 */
100char* get_current_branch() {
101 // Execute 'git rev-parse' command to get the branch name
102 FILE* fp = popen("git rev-parse --abbrev-ref HEAD 2>/dev/null", "r");
103 if (fp == NULL) {
104 return NULL; // Return NULL if command execution failed
105 }
106
107 char* branch = malloc(256); // Allocate memory for the branch name
108 if (fgets(branch, 256, fp) == NULL) { // Read the branch name
109 free(branch); // Free allocated memory if read failed
110 pclose(fp); // Close the command pipe
111 return NULL; // Return NULL to indicate failure
112 }
113
114 pclose(fp); // Close the command pipe
115 branch[strcspn(branch, "\n")] = 0; // Remove newline character
116 return branch; // Return the branch name
117}
