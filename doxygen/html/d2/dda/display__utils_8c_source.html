35/**
36 * @brief Determines the byte width of a UTF-8 character.
37 *
38 * This function analyzes the first byte of a UTF-8 character to determine
39 * its total byte width.
40 *
41 * @param str Pointer to the start of a UTF-8 character.
42 * @return The number of bytes used by the character (1-4).
43 */
44size_t get_utf8_char_width(const char *str) {
45 unsigned char c = (unsigned char)*str; // Get the first byte of the character
46 if (c < 0x80) return 1; // ASCII character (0xxxxxxx)
47 if (c < 0xE0) return 2; // 2-byte UTF-8 character (110xxxxx)
48 if (c < 0xF0) return 3; // 3-byte UTF-8 character (1110xxxx)
49 return 4; // 4-byte UTF-8 character (11110xxx)
50}

52/**
53 * @brief Calculates the display width of a string, accounting for multi-byte characters.
54 *
55 * This function iterates through the string, determining the width of each character
56 * and summing them to get the total display width.
57 *
58 * @param str The string to measure.
59 * @return The display width of the string.
60 */
61size_t get_display_width(const char *str) {
62 size_t width = 0;
63 while (*str) { // Loop through each character in the string
64 size_t char_width = get_utf8_char_width(str); // Get the byte width of the current character
65 if (char_width == 4) {
66 width += 2; // Emoji and other wide characters count as 2 display units
67 } else if (char_width > 1) {
68 width += 1; // Other multi-byte characters count as 1 display unit
69 } else {
70 width += 1; // ASCII characters count as 1 display unit
71 }
72 str += char_width; // Move to the next character
73 }
74 return width;
75}

118 free(branch);

120 printf("\033[1m%s\033[0m\n", current_dir);

124 // Allocate memory for storing entry widths
125 size_t *entry_widths = malloc(num_entries * sizeof(size_t));
126 if (!entry_widths) {
127 fprintf(stderr, "Memory allocation failed\n");
128 return;
129 }
130
131 // Calculate the width of each entry and find the maximum width
132 size_t max_width = 0;
133 for (int i = 0; i < num_entries; i++) {
134 // Calculate width: emoji + spacing + name + potential Git status
135 entry_widths[i] = EMOJI_WIDTH + EMOJI_NAME_SPACING + get_display_width(entries[i].name);
136 if (entries[i].git_status[0] != '\0') {
137 entry_widths[i] += 3; // Add 3 for "(X)" Git status
138 }
139 if (entry_widths[i] > max_width) {
140 max_width = entry_widths[i];
141 }
142 }
143
144 // Calculate the number of columns based on terminal width
145 int num_columns = (term_width + SPACING) / (max_width + SPACING);
146 int max_columns = get_max_columns(current_dir);
147 if (num_columns > max_columns) num_columns = max_columns;
148 if (num_columns < 1) num_columns = 1;
149
150 // Allocate memory for column widths
151 size_t *column_widths = calloc(num_columns, sizeof(size_t));
152 if (!column_widths) {
153 fprintf(stderr, "Memory allocation failed\n");
154 free(entry_widths);
155 return;
156 }
157
158 // Calculate the number of rows
159 int rows = (num_entries + num_columns - 1) / num_columns;
160
161 // Calculate the width of each column
162 for (int i = 0; i < num_entries; i++) {
163 int col = i / rows;
164 if (entry_widths[i] > column_widths[col]) {
165 column_widths[col] = entry_widths[i];
166 }
167 }
168
169 // Display the entries in a grid format
170 for (int row = 0; row < rows; row++) {
171 for (int col = 0; col < num_columns; col++) {
172 int index = col * rows + row;
173 if (index < num_entries) {
174 // Print emoji
175 printf("%s ", entries[index].emoji);
176 // Print file name
177 printf("%s", entries[index].name);
178
179 // Print Git status if available

181 //printf("(\033[32m%c\033[0m)", entries[index].git_status[0]);
182 printf("\033[38;5;202m[%c]\033[0m", entries[index].git_status[0]);
183 }
184
185 // Calculate and print padding
186 size_t printed_width = entry_widths[index];
187 size_t padding = (col < num_columns - 1) ? column_widths[col] - printed_width + SPACING : 0;
188 for (size_t i = 0; i < padding; i++) {
189 printf(" ");
190 }
191 }
192 }
193 printf("\n"); // Move to the next row
194 }
