59/**
60 * @brief Formats a file size into a human-readable string.
61 *
62 * @param size The size in bytes to format.
63 * @return A pointer to a static buffer containing the formatted size string.
64 */
65static char *format_size(off_t size) {
66 static char buf[64];
67 const char *units[] = {"B", "K", "M", "G", "T", "P", "E", "Z", "Y"};
68 int i = 0;
69 double dsize = size;
70
71 while (dsize >= 1024 && i < 8) {
72 dsize /= 1024;
73 i++;
74 }
75
76 snprintf(buf, sizeof(buf), "%.1f%s", dsize, units[i]);
77 return buf;
78}

80/**
81 * @brief Formats the time difference between now and a given time into a human-readable string.
82 *
83 * @param file_time The time to format.
84 * @param buf Buffer to store the formatted string.
85 * @param buf_size Size of the buffer.
86 */
87static void format_time_ago(time_t file_time, char *buf, size_t buf_size) {
88 time_t now = time(NULL);
89 double seconds = difftime(now, file_time);
90 int minutes = (int)(seconds / 60);
91 int hours = minutes / 60;
92 int days = hours / 24;
93 int months = days / 30; // Approximation
94 int years = days / 365; // Approximation
95
96 if (years > 0) {
97 snprintf(buf, buf_size, "%dy %dm ago", years, months % 12);
98 } else if (months > 0) {
99 snprintf(buf, buf_size, "%dm %dd ago", months, days % 30);
100 } else if (days > 0) {
101 snprintf(buf, buf_size, "%dd ago", days);
102 } else if (hours > 0) {
103 snprintf(buf, buf_size, "%dh ago", hours);
104 } else if (minutes > 0) {
105 snprintf(buf, buf_size, "%dm ago", minutes);
106 } else {
107 snprintf(buf, buf_size, "just now");
108 }
109}

112 * @brief Converts file permissions to a string of emojis representing user rights.
113 *
114 * @param mode The file mode containing the permissions.
115 * @param rights Buffer to store the emoji string.
116 */
117static void get_user_rights(mode_t mode, char *rights) {
118 const char *read = "\xF0\x9F\x91\x80"; // 👀
119 const char *write = "\xE2\x9C\x8F\xEF\xB8\x8F"; // ✏️
120 const char *exec = "\xF0\x9F\x9A\x80"; // 🚀
121 const char *no = "\xE2\x9D\x8C"; // ❌
122
123 strcpy(rights, "");
124 strcat(rights, (mode & S_IRUSR) ? read : no);
125 strcat(rights, (mode & S_IWUSR) ? write : no);
126 strcat(rights, (mode & S_IXUSR) ? exec : no);
127}

129/**
130 * @brief Calculates the total size of a directory and its contents.
131 *
132 * @param path The path of the directory.
133 * @return The total size of the directory in bytes.
134 */
135static off_t get_dir_size(const char *path) {
136 DIR *dir;
137 struct dirent *entry;
138 struct stat st;
139 off_t total_size = 0;
140 char full_path[MAX_PATH];
141
142 dir = opendir(path);
143 if (!dir) return 0;
144
145 while ((entry = readdir(dir)) != NULL) {
146 if (strcmp(entry->d_name, ".") == 0 || strcmp(entry->d_name, "..") == 0) continue;
147
148 snprintf(full_path, sizeof(full_path), "%s/%s", path, entry->d_name);
149
150 if (lstat(full_path, &st) == 0) {
151 if (S_ISDIR(st.st_mode)) {
152 total_size += get_dir_size(full_path);
153 } else {
154 total_size += st.st_size;
155 }
156 }
157 }
158 closedir(dir);
159 return total_size;
160}

191/**
192 * @brief Comparison function for sorting file_info structures.
193 *
194 * @param a Pointer to the first file_info structure.
195 * @param b Pointer to the second file_info structure.
196 * @return Integer less than, equal to, or greater than zero if a is found,
197 * respectively, to be less than, to match, or be greater than b.
198 */
199static int compare_entries(const void *a, const void *b) {
200 const struct file_info *fa = (const struct file_info *)a;
201 const struct file_info *fb = (const struct file_info *)b;
202
203 if (fa->is_dir && !fb->is_dir) return -1;
204 if (!fa->is_dir && fb->is_dir) return 1;
205
206 if (fa->is_dir && fb->is_dir) {
207 if (fa->subdir_count != fb->subdir_count) {
208 return fb->subdir_count - fa->subdir_count;
209 }
210 if (fa->size != fb->size) {
211 return (fa->size < fb->size) ? 1 : -1;
212 }
213 }
214
215 if (!fa->is_dir && !fb->is_dir) {
216 if (fa->size != fb->size) {
217 return (fa->size < fb->size) ? 1 : -1;
218 }
219 }
220
221 return strcasecmp(fa->name, fb->name);
222}

224/**
225 * @brief Gets detailed information about a file.
226 *
227 * @param path The path to the file.
228 * @param fi Pointer to a file_info structure to populate.
229 * @param max_owner_len Pointer to store the maximum owner name length.
230 * @param max_time_ago_len Pointer to store the maximum time ago string length.
231 * @return 0 on success, -1 on failure.
232 */
233static int get_file_info(const char *path, struct file_info *fi, size_t *max_owner_len, size_t *max_time_ago_len) {
234 struct stat st;
235 if (lstat(path, &st) != 0) return -1;
236
237 strncpy(fi->name, path, sizeof(fi->name) - 1);
238 fi->name[sizeof(fi->name) - 1] = '\0';
239 strncpy(fi->full_path, path, sizeof(fi->full_path) - 1);
240 fi->full_path[sizeof(fi->full_path) - 1] = '\0';
241
242 fi->mode = st.st_mode;
243 fi->mtime = st.st_mtime;
244 fi->is_dir = S_ISDIR(st.st_mode);
245 fi->size = fi->is_dir ? get_dir_size(path) : st.st_size;
246 fi->subdir_count = fi->is_dir ? count_subdirs(path) : 0;
247
248 struct passwd *pw = getpwuid(st.st_uid);
249 if (pw) {
250 strncpy(fi->owner, pw->pw_name, sizeof(fi->owner) - 1);
251 } else {
252 snprintf(fi->owner, sizeof(fi->owner), "%d", st.st_uid);
253 }
254
255 size_t owner_len = strlen(fi->owner);
256 if (owner_len > *max_owner_len) {
257 *max_owner_len = owner_len;
258 }
259
260 format_time_ago(fi->mtime, fi->time_ago, sizeof(fi->time_ago));
261 size_t time_ago_len = strlen(fi->time_ago);
262 if (time_ago_len > *max_time_ago_len) {
263 *max_time_ago_len = time_ago_len;
264 }
265
266 get_user_rights(fi->mode, fi->user_rights);
267
268 return 0;
269}

271/**
272 * @brief Prints sorted entries in a detailed format.
273 *
274 * @param entries Array of file_info structures.
275 * @param entry_count Number of entries in the array.
276 * @param max_owner_len Maximum length of owner names.
277 * @param max_time_ago_len Maximum length of time ago strings.

279static void print_sorted_entries(struct file_info *entries, int entry_count, size_t max_owner_len, size_t max_time_ago_len) {
280 for (int i = 0; i < entry_count; i++) {
281 struct file_info *fi = &entries[i];
282 char *emoji = get_emoji(fi->full_path);
283
284 printf("%8s %-*s \xF0\x9F\x91\x91: %-*s \xF0\x9F\x93\x9C: %s %s %-17s",
285 format_size(fi->size),
286 (int)max_time_ago_len, fi->time_ago,
287 (int)max_owner_len, fi->owner,
288 fi->user_rights,
289 emoji,
290 fi->name);
291
292 if (fi->is_dir) {
293 printf("(%d)", fi->subdir_count);
294 }
295 printf("\n");
296
297 free(emoji);
298 }

301/**
302 * @brief Prints a detailed listing of the contents of a directory.
303 *
304 * @param path The path of the directory to list.
305 */
306void print_longlisting(const char *path) {
307 setlocale(LC_ALL, "");
308
309 struct file_info entries[MAX_ENTRIES];
310 int entry_count = 0;
311 size_t max_owner_len = 0;
312 size_t max_time_ago_len = 0;
313
314 DIR *dir = opendir(path);
315 if (!dir) {
316 perror("Error opening directory");
317 return;
318 }
319
320 struct dirent *entry;
321 while ((entry = readdir(dir)) != NULL && entry_count < MAX_ENTRIES) {
322 if (strcmp(entry->d_name, ".") == 0 || strcmp(entry->d_name, "..") == 0) continue;
323
324 char full_path[MAX_PATH];
325 snprintf(full_path, sizeof(full_path), "%s/%s", path, entry->d_name);
326
327 struct file_info *fi = &entries[entry_count];
328 if (get_file_info(full_path, fi, &max_owner_len, &max_time_ago_len) == 0) {
329 entry_count++;
330 }
331 }
332 closedir(dir);
333
334 qsort(entries, entry_count, sizeof(struct file_info), compare_entries);
335 print_sorted_entries(entries, entry_count, max_owner_len, max_time_ago_len);
336}

339void print_longlisting_files(const char **patterns, int pattern_count) {
340 struct file_info entries[MAX_ENTRIES];
341 int entry_count = 0;
342 size_t max_owner_len = 0;
343 size_t max_time_ago_len = 0;
344
345 for (int i = 0; i < pattern_count; i++) {
346 glob_t globbuf;
347 int glob_result = glob(patterns[i], GLOB_TILDE, NULL, &globbuf);
348
349 if (glob_result == 0) {
350 for (size_t j = 0; j < globbuf.gl_pathc && entry_count < MAX_ENTRIES; j++) {
351 struct file_info *fi = &entries[entry_count];
352 if (get_file_info(globbuf.gl_pathv[j], fi, &max_owner_len, &max_time_ago_len) == 0) {
353 entry_count++;
354 }
355 }
356 } else {
357 fprintf(stderr, "glob error for pattern '%s': %s\n", patterns[i], strerror(errno));
358 }
359 globfree(&globbuf);
360 }
361
362 if (entry_count > 0) {
363 qsort(entries, entry_count, sizeof(struct file_info), compare_entries);
364 print_sorted_entries(entries, entry_count, max_owner_len, max_time_ago_len);
365 } else {
366 fprintf(stderr, "No matching files found.\n");
367 }
