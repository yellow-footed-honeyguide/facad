33/**
34 * @brief Parses command-line arguments.
35 *
36 * This function processes the command-line arguments, setting appropriate flags
37 * in the CommandLineArgs structure and collecting non-option arguments as targets.
38 *
39 * @param argc The number of command-line arguments.
40 * @param argv An array of strings containing the command-line arguments.
41 * @return A CommandLineArgs structure containing the parsed arguments.
42 */
43CommandLineArgs parse_args(int argc, char *argv[]) {
44 CommandLineArgs args = {0}; // Initialize the args structure with zeros
45 args.targets = malloc(sizeof(char*) * (argc - 1)); // Allocate memory for targets
46 args.target_count = 0; // Initialize target count to 0
47
48 // Loop through all arguments starting from index 1 (skip program name)
49 for (int i = 1; i < argc; i++) {
50 if (strcmp(argv[i], "-v") == 0 || strcmp(argv[i], "--version") == 0) {
51 args.show_version = true; // Set flag to show version
52 } else if (strcmp(argv[i], "-h") == 0 || strcmp(argv[i], "--help") == 0) {
53 args.show_help = true; // Set flag to show help
54 } else if (strcmp(argv[i], "-l") == 0 || strcmp(argv[i], "--long") == 0) {
55 args.show_longlisting = true; // Set flag for long listing
56 } else if (strcmp(argv[i], "-a") == 0 || strcmp(argv[i], "--analytics") == 0) {
57 args.show_dir_analytics = true; // Set flag for directory analytics
58 } else if (argv[i][0] == '-') {
59 args.invalid_opt = argv[i]; // Store invalid option
60 return args; // Return immediately on invalid option
61 } else {
62 args.targets[args.target_count++] = argv[i]; // Store non-option argument as target
63 }
64 }
65
66 return args; // Return the populated CommandLineArgs structure
67}

77void free_args(CommandLineArgs *args) {
