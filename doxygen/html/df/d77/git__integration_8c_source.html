20/**
21 * @brief Integrates Git status information into file entries.
22 *
23 * This function retrieves the Git status for files in the current directory
24 * and updates the corresponding FileCardInfo structures with this information.
25 *
26 * @param entries Pointer to an array of FileCardInfo pointers.
27 * @param num_entries Pointer to the number of entries in the array.
28 * @param current_dir The current directory path.
29 */
30void integrate_git_status(FileCardInfo** entries, int* num_entries, const char* current_dir) {
31 // Check if the current directory is a Git repository
32 if (!is_git_repository()) {
33 return; // If not, return without doing anything
34 }
35
36 // Get the Git status for all files in the repository
37 GitFileList git_status = get_git_status();
38
39 // Create a copy of the current directory path
40 char* base_dir = strdup(current_dir);
41 // Extract the base name of the directory
42 char* dir_name = basename(base_dir);
43
44 // Iterate through all file entries
45 for (int i = 0; i < *num_entries; i++) {
46 // Initialize Git status as empty for each entry
47 (*entries)[i].git_status[0] = '\0';
48 (*entries)[i].git_status[1] = '\0';
49
50 // Construct the relative path for the current file
51 char relative_path[MAX_FILENAME];
52 snprintf(relative_path, sizeof(relative_path), "%s/%s", dir_name, (*entries)[i].name);
53
54 // Compare the file's path with Git status entries
55 for (int j = 0; j < git_status.count; j++) {
56 // If the paths match, update the Git status in the file entry
57 if (strcmp(relative_path, git_status.files[j].filename) == 0 ||
58 strcmp((*entries)[i].name, git_status.files[j].filename) == 0) {
59 (*entries)[i].git_status[0] = git_status.files[j].status;
60 (*entries)[i].git_status[1] = '\0';
61 break; // Stop searching once a match is found
62 }
63 }
64 }
65
66 // Free the memory allocated for the base directory string
67 free(base_dir);
68}
