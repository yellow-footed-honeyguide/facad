35/**
36 * @brief Formats a file size into a human-readable string.
37 *
38 * @param size The size in bytes to format.
39 * @return A pointer to a static buffer containing the formatted size string.
40 */
41static char* format_size(off_t size) {
42 static char buf[64];
43 const char* units[] = {"B", "K", "M", "G", "T", "P", "E", "Z", "Y"};
44 int i = 0;
45 double dsize = size;
46
47 // Convert size to appropriate unit
48 while (dsize >= 1024 && i < 8) {
49 dsize /= 1024;
50 i++;
51 }
52
53 // Format the size with one decimal place
54 snprintf(buf, sizeof(buf), "%.1f%s", dsize, units[i]);
55 return buf;
56}

71/**
72 * @brief Recursively scans a directory to find its maximum depth.
73 *
74 * @param path The path of the directory to scan.
75 * @param max_depth Pointer to the current maximum depth.
76 * @param current_depth The current depth in the recursion.
77 * @param deepest_dir Buffer to store the path of the deepest directory.
78 */
79static void recursive_dir_scan(const char* path, int* max_depth, int current_depth, char* deepest_dir) {
80 DIR* dir;
81 struct dirent* entry;
82 char full_path[MAX_PATH];
83
84 dir = opendir(path);
85 if (!dir) return;
86
87 while ((entry = readdir(dir)) != NULL) {
88 // Skip . and .. entries
89 if (strcmp(entry->d_name, ".") == 0 || strcmp(entry->d_name, "..") == 0) continue;
90
91 snprintf(full_path, sizeof(full_path), "%s/%s", path, entry->d_name);
92
93 struct stat st;
94 if (lstat(full_path, &st) == 0 && S_ISDIR(st.st_mode)) {
95 int new_depth = current_depth + 1;
96 // Update max_depth and deepest_dir if a new maximum is found
97 if (new_depth > *max_depth) {
98 *max_depth = new_depth;
99 strncpy(deepest_dir, full_path, MAX_PATH - 1);
100 deepest_dir[MAX_PATH - 1] = '\0';
101 }
102 // Recursively scan subdirectory
103 recursive_dir_scan(full_path, max_depth, new_depth, deepest_dir);
104 }
105 }
106 closedir(dir);
107}

109/**
110 * @brief Calculates the total size of a directory and its contents.
111 *
112 * @param path The path of the directory to analyze.
113 * @return The total size of the directory in bytes.
114 */
115static off_t calculate_total_size(const char* path) {
116 DIR* dir;
117 struct dirent* entry;
118 struct stat st;
119 off_t total_size = 0;
120 char full_path[MAX_PATH];
121 static long block_size = 0;
122
123 // Determine the filesystem block size
124 if (block_size == 0) {
125 block_size = 512; // Default value
126#ifdef ST_BLKSIZE
127 struct statvfs buf;
128 if (statvfs(path, &buf) == 0) {
129 block_size = buf.f_frsize;
130 }
131#endif
132 }
133
134 dir = opendir(path);
135 if (!dir) {
136 fprintf(stderr, "Error opening directory %s: %s\n", path, strerror(errno));
137 return 0;
138 }
139
140 while ((entry = readdir(dir)) != NULL) {
141 if (strcmp(entry->d_name, ".") == 0 || strcmp(entry->d_name, "..") == 0) continue;
142
143 snprintf(full_path, sizeof(full_path), "%s/%s", path, entry->d_name);
144
145 if (lstat(full_path, &st) == 0) {
146 if (S_ISDIR(st.st_mode)) {
147 // Recursively calculate size for subdirectories
148 total_size += calculate_total_size(full_path);
149 } else if (S_ISREG(st.st_mode)) {
150 // Add file size, rounded up to the nearest block
151 total_size += (st.st_blocks * 512 + block_size - 1) / block_size * block_size;
152 }
153 // Ignore symbolic links and other special files
154 } else {
155 fprintf(stderr, "Error stating %s: %s\n", full_path, strerror(errno));
156 }
157 }
158 closedir(dir);
159 return total_size;
160}

202/**
203 * @brief Prints detailed analytics about a directory.
204 *
205 * This function analyzes the contents of the specified directory
206 * and prints various statistics including total size, file counts,
207 * directory depth, and information about largest and newest files.
208 *
209 * @param path The path of the directory to analyze.
210 */
211void print_dir_analytics(const char* path) {
212 DIR* dir;
213 struct dirent* entry;
214 struct stat st;
215 off_t largest_size = 0;
216 int total_files = 0, total_dirs = 0, empty_files = 0, max_depth = 0;
217 time_t newest_time = 0, oldest_time = time(NULL);
218 char largest_file[MAX_PATH] = "", newest_file[MAX_PATH] = "", oldest_file[MAX_PATH] = "";
219 char empty_file_names[MAX_FILES][256];
220 char deepest_dir[MAX_PATH] = "";
221
222 dir = opendir(path);
223 if (!dir) {
224 perror("Error opening directory");
225 return;
226 }
227
228 // Find the largest file recursively
229 find_largest_file(path, &largest_size, largest_file, sizeof(largest_file));
230
231 // Calculate total size recursively
232 off_t total_size = calculate_total_size(path);
233
234 while ((entry = readdir(dir)) != NULL) {
235 char full_path[MAX_PATH];
236 snprintf(full_path, sizeof(full_path), "%s/%s", path, entry->d_name);
237 if (lstat(full_path, &st) != 0) continue;
238
239 if (strcmp(entry->d_name, ".") == 0 || strcmp(entry->d_name, "..") == 0) continue;
240
241 if (S_ISDIR(st.st_mode)) {
242 total_dirs++;
243 } else if (S_ISREG(st.st_mode)) {
244 total_files++;
245
246 // Check for largest file
247 if (st.st_size > largest_size) {
248 largest_size = st.st_size;
249 strncpy(largest_file, entry->d_name, sizeof(largest_file) - 1);
250 }
251
252 // Check for newest file
253 if (st.st_mtime > newest_time) {
254 newest_time = st.st_mtime;
255 strncpy(newest_file, entry->d_name, sizeof(newest_file) - 1);
256 }
257
258 // Check for oldest file
259 if (st.st_mtime < oldest_time) {
260 oldest_time = st.st_mtime;
261 strncpy(oldest_file, entry->d_name, sizeof(oldest_file) - 1);
262 }
263
264 // Check for empty files
265 if (st.st_size == 0) {
266 strncpy(empty_file_names[empty_files], entry->d_name, 255);
267 empty_file_names[empty_files][255] = '\0';
268 empty_files++;
269 }
270 }
271 }
272 closedir(dir);
273
274 // Calculate max depth and find deepest directory
275 recursive_dir_scan(path, &max_depth, 0, deepest_dir);
276
277 char cwd[MAX_PATH];
278 if (getcwd(cwd, sizeof(cwd)) == NULL) {
279 perror("getcwd() error");
280 return;
281 }
282
283 // Print analytics
284 printf("%s%s%s\n", ANSI_BOLD, cwd, ANSI_RESET);
285 printf(" Total Size : %s\n", format_size(total_size));
286 printf(" Directories : %d\n", total_dirs);
287 printf(" Max Depth : %d levels\n", max_depth);
288 printf(" Deepest Dir : %s\n", deepest_dir);
289 printf(" Files : %d\n", total_files);
290 printf(" Largest File : %s [%s]\n", largest_file, format_size(largest_size));
291 printf(" Oldest File : %s [%s]\n", oldest_file, format_time(oldest_time));
292 printf(" Newest File : %s [%s]\n", newest_file, format_time(newest_time));
293 printf(" Empty Files : %d [", empty_files);
294
295 // Print names of empty files
296 for (int i = 0; i < empty_files; i++) {
297 printf("%s%s", empty_file_names[i], (i + 1 < empty_files) ? " " : "");
298 }
299 printf("]\n");
300}
