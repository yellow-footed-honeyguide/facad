37/**
38 * @brief Dynamically allocates or reallocates memory for directory entries.
39 *
40 * This function manages the memory for the array of FileCardInfo structures.
41 * It starts with INITIAL_ENTRIES and doubles the size when more space is needed.
42 *
43 * @param entries Pointer to the array of FileCardInfo structures.
44 * @param current_size Pointer to the current size of the array.
45 * @return 0 on success, -1 on failure.
46 */
47static int allocate_entries(FileCardInfo **entries, int *current_size) {
48 // Calculate new size (double the current size or use initial size if NULL)
49 int new_size = (*entries == NULL) ? INITIAL_ENTRIES : (*current_size * 2);
50
51 // Attempt to reallocate memory for entries
52 FileCardInfo *new_entries = realloc(*entries, new_size * sizeof(FileCardInfo));
53
54 // Check if reallocation was successful
55 if (new_entries == NULL) {
56 perror("Failed to allocate memory for entries"); // Print error message
57 return -1; // Return -1 to indicate failure
58 }
59
60 // Update entries pointer and current size
61 *entries = new_entries;
62 *current_size = new_size;
63 return 0; // Return 0 to indicate success
64}

66/**
67 * @brief Processes directory entries and populates the entries array.
68 *
69 * This function reads directory contents, creates FileCardInfo structures for each entry,
70 * and dynamically allocates memory as needed. It also integrates Git status information.
71 *
72 * @param dir_path Path to the directory to process.
73 * @param entries Pointer to the array of FileCardInfo structures.
74 * @param num_entries Pointer to the number of entries processed.
75 * @param current_size Pointer to the current size of the entries array.
76 * @return 0 on success, -1 on failure.
77 */
78static int process_directory(const char *dir_path, FileCardInfo **entries, int *num_entries, int *current_size) {
79 DIR *dir = opendir(dir_path); // Open the directory
80 if (dir == NULL) {
81 perror("Error opening directory"); // Print error message if directory can't be opened
82 return -1;
83 }
84
85 struct dirent *entry;
86 while ((entry = readdir(dir)) != NULL) { // Loop through each entry in the directory
87 // Skip '.' and '..' entries
88 if (strcmp(entry->d_name, ".") == 0 || strcmp(entry->d_name, "..") == 0) {
89 continue;
90 }
91
92 // Check if we need to allocate more memory
93 if (*num_entries >= *current_size) {
94 if (allocate_entries(entries, current_size) != 0) {
95 closedir(dir); // Close the directory before returning
96 return -1;
97 }
98 }
99
100 // Construct full path for the current entry
101 char full_path[MAX_PATH];
102 snprintf(full_path, sizeof(full_path), "%s/%s", dir_path, entry->d_name);
103
104 // Create a file entry and add it to the array
105 if (create_file_entry(&(*entries)[*num_entries], entry->d_name) != 0) {
106 fprintf(stderr, "Warning: Unable to get info for %s: %s\n", full_path, strerror(errno));
107 continue;
108 }
109
110 (*num_entries)++; // Increment the number of entries
111 }
112
113 closedir(dir); // Close the directory
114
115 // Integrate Git status information for the entries
116 integrate_git_status(entries, num_entries, dir_path);
117
118 return 0; // Return 0 to indicate success
119}

121/**
122 * @brief Processes files or patterns and populates the entries array.
123 *
124 * This function handles file patterns (including globs) and individual files,
125 * creating FileCardInfo structures for matching entries.
126 *
127 * @param patterns Array of file patterns or names to process.
128 * @param pattern_count Number of patterns in the array.
129 * @param entries Pointer to the array of FileCardInfo structures.
130 * @param num_entries Pointer to the number of entries processed.
131 * @param current_size Pointer to the current size of the entries array.
132 * @return 0 on success, -1 on failure.
133 */
134static int process_files_or_patterns(const char **patterns, int pattern_count, FileCardInfo **entries, int *num_entries, int *current_size) {
135 DIR *dir;
136 struct dirent *entry;
137 char full_path[MAX_PATH];
138
139 dir = opendir("."); // Open the current directory
140 if (dir == NULL) {
141 perror("Error opening current directory"); // Print error message if directory can't be opened
142 return -1;
143 }
144
145 while ((entry = readdir(dir)) != NULL) { // Loop through each entry in the directory
146 bool match = false;
147 // Check if the entry matches any of the provided patterns
148 for (int i = 0; i < pattern_count; i++) {
149 if (fnmatch(patterns[i], entry->d_name, 0) == 0) {
150 match = true;
151 break;
152 }
153 }
154
155 if (match) { // If the entry matches a pattern
156 // Check if we need to allocate more memory
157 if (*num_entries >= *current_size) {
158 if (allocate_entries(entries, current_size) != 0) {
159 closedir(dir); // Close the directory before returning
160 return -1;
161 }
162 }
163
164 // Create a file entry for the matching file
165 snprintf(full_path, sizeof(full_path), "%s", entry->d_name);
166 if (create_file_entry(&(*entries)[*num_entries], full_path) != 0) {
167 fprintf(stderr, "Warning: Unable to get info for %s: %s\n", full_path, strerror(errno));
168 continue;
169 }
170
171 (*num_entries)++; // Increment the number of entries
172 }
173 }
174
175 closedir(dir); // Close the directory
176 return *num_entries > 0 ? 0 : -1; // Return 0 if entries were found, -1 otherwise
177}

185static int is_directory(const char *path) {

192/**
193 * @brief Processes a single target (directory, file, or pattern).
194 *
195 * This function handles different types of targets: directories, glob patterns,
196 * and individual files. It populates the entries array accordingly.
197 *
198 * @param target The target to process (directory path, file path, or pattern).
199 * @param entries Pointer to the array of FileCardInfo structures.
200 * @param num_entries Pointer to the number of entries processed.
201 * @param current_size Pointer to the current size of the entries array.
202 * @return 0 on success, -1 on failure.
203 */
204static int process_target(const char *target, FileCardInfo **entries, int *num_entries, int *current_size) {
205 if (is_directory(target)) { // If the target is a directory
206 char previous_dir[MAX_PATH];
207 if (getcwd(previous_dir, sizeof(previous_dir)) == NULL) { // Get current working directory
208 perror("getcwd() error");
209 return -1;
210 }
211
212 if (chdir(target) != 0) { // Change to the target directory
213 fprintf(stderr, "Error changing to directory '%s': %s\n", target, strerror(errno));
214 return -1;
215 }
216
217 int result = process_directory(".", entries, num_entries, current_size); // Process the directory
218
219 // Change back to the previous directory
220 if (chdir(previous_dir) != 0) {
221 fprintf(stderr, "Error changing back to previous directory: %s\n", strerror(errno));
222 return -1;
223 }
224
225 return result;
226 } else if (is_glob_pattern(target)) { // If the target is a glob pattern
227 return process_files_or_patterns(&target, 1, entries, num_entries, current_size);
228 } else { // Assume it's a single file
229 // Check if we need to allocate more memory
230 if (*num_entries >= *current_size) {
231 if (allocate_entries(entries, current_size) != 0) {
232 return -1;
233 }
234 }
235 // Create a file entry for the single file
236 if (create_file_entry(&(*entries)[*num_entries], target) != 0) {
237 fprintf(stderr, "Unable to get info for %s: %s\n", target, strerror(errno));
238 return -1;
239 }
240 (*num_entries)++; // Increment the number of entries
241 }
242 return 0; // Return 0 to indicate success
243}

248 * This function determines whether the given targets include glob patterns
249 * or specific files, as opposed to just directories.

253 * @return 1 if glob patterns or specific files are present, 0 otherwise.

255static int is_glob_or_specific_files(const char **targets, int target_count) {
256 for (int i = 0; i < target_count; i++) {
257 if (is_glob_pattern(targets[i]) || !is_directory(targets[i])) {
258 return 1; // Return 1 if a glob pattern or non-directory is found

264/**

266 *
267 * This function serves as the entry point for the facad directory listing tool.
268 * It parses command-line arguments, processes directories or files based on the input,
269 * and displays the results using the appropriate format (standard, long listing, or analytics).

272 * @param argv Array of command-line argument strings.

284

291
292 // Handle invalid options

299

306 }
307
308 // Handle long listing request
309 if (args.show_longlisting) {

311 print_longlisting_files(args.targets, args.target_count);

317 }
318
319 // Handle directory analytics request
320 if (args.show_dir_analytics) {
321 print_dir_analytics(args.target_count > 0 ? args.targets[0] : ".");
322 free_args(&args);
323 return EXIT_SUCCESS;
324 }
325
326 // Get terminal width for display formatting
327 struct winsize w;
328 ioctl(STDOUT_FILENO, TIOCGWINSZ, &w);
329 int term_width = w.ws_col;
330
331 // Initialize variables for storing file entries
332 FileCardInfo *entries = NULL;
333 int num_entries = 0;
334 int current_size = 0;
335
336 // Set up display path
337 char display_path[MAX_PATH];
338 strcpy(display_path, current_dir);
339
340 int show_path = 1; // Flag to determine if path should be displayed
341
342 // Process targets if provided, otherwise process current directory
343 if (args.target_count > 0) {
344 show_path = 0; // Don't show path for specific targets
345 for (int i = 0; i < args.target_count; i++) {
346 // Get real path if possible
347 char *real_path = realpath(args.targets[i], NULL);
348 if (real_path != NULL) {
349 strcpy(display_path, real_path);
350 free(real_path);
351 } else {
352 strcpy(display_path, args.targets[i]);
353 }
354
355 // Process the target
356 if (process_target(args.targets[i], &entries, &num_entries, &current_size) != 0) {
357 fprintf(stderr, "Error processing '%s'\n", args.targets[i]);
358 goto cleanup; // Jump to cleanup if there's an error
359 }
360 }
361 } else {

363 if (is_dev_directory(current_dir)) {
364 strcpy(display_path, "/dev");
365 if (handle_dev_directory(&entries, &num_entries, &current_size) != 0) {
366 fprintf(stderr, "Error processing /dev directory\n");
367 goto cleanup;
368 }
369 } else {
370 // Process current directory
371 if (process_directory(".", &entries, &num_entries, &current_size) != 0) {
372 fprintf(stderr, "Error processing current directory\n");
373 goto cleanup;
374 }
375 }
376 }
377
378 // Handle case when no entries are found
379 if (num_entries == 0) {
380 char current_dir[PATH_MAX];
381 if (getcwd(current_dir, sizeof(current_dir)) != NULL) {
382 // Display current directory if it matches the display path
383 if (strcmp(current_dir, display_path) == 0) {
384 printf("\033[1m%s\033[0m\n", current_dir);
385 }
386 fprintf(stderr, "🚫 No files found\n");
387 } else {
388 fprintf(stderr, "🚫 No files found\n");
389 }
390 } else {
391 // Sort the entries alphabetically
392 qsort(entries, num_entries, sizeof(FileCardInfo), compare_file_entries);
393
394 // If in a Git repository, integrate Git status information
395 if (is_git_repository()) {
396 GitFileList git_status = get_git_status();
397 for (int i = 0; i < num_entries; i++) {
398 for (int j = 0; j < git_status.count; j++) {
399 // Match file names and set Git status
400 if (strcmp(entries[i].name, git_status.files[j].filename) == 0) {
401 entries[i].git_status[0] = git_status.files[j].status;
402 entries[i].git_status[1] = '\0';
403 break;
404 }
405 }
406 }
407 }
408
409 // Display the entries using the appropriate format
410 display_entries(entries, num_entries, term_width, display_path, show_path);
411 }
412
413cleanup:
414 // Free allocated memory for each entry
415 for (int i = 0; i < num_entries; i++) {
416 free_file_entry(&entries[i]);
417 }
418 free(entries); // Free the main entries array
419 free_args(&args); // Free command-line arguments
420
421 // Return success if entries were found, failure otherwise
422 return num_entries > 0 ? EXIT_SUCCESS : EXIT_FAILURE;
423}
