89/**
90 * @brief Determines the appropriate emoji for a given file based on its characteristics
91 *
92 * @param path The path to the file
93 * @return A dynamically allocated string containing the emoji
94 */
95char *get_emoji(const char *path) {
96 struct stat path_stat;
97
98 // Check if we can get file information
99 if (lstat(path, &path_stat) != 0) {
100 return safe_strdup("❓"); // Unknown file type
101 }
102
103 // Check for symbolic links
104 if (S_ISLNK(path_stat.st_mode)) {
105 return safe_strdup(S_ISDIR(path_stat.st_mode) ? "🔗📁" : "🔗");
106 }
107
108 // Check for directories
109 if (S_ISDIR(path_stat.st_mode)) {
110 return safe_strdup("📁");
111 }
112
113 // Extract the filename from the path
114 const char *filename = strrchr(path, '/');
115 filename = filename ? filename + 1 : path;
116
117 // Check for exact file name matches first (highest priority)
118 for (size_t i = 0; i < emoji_exact_file_map_size; i++) {
119 if (strcasecmp(filename, emoji_exact_file_map[i].key) == 0) {
120 return safe_strdup(emoji_exact_file_map[i].emoji);
121 }
122 }
123
124 // Check file content for specific patterns (second priority)
125 char *content_emoji = check_file_content(path);
126 if (content_emoji) {
127 return content_emoji;
128 }
129
130 // Check file extensions (third priority)
131 char *extension = strrchr(filename, '.');
132 if (extension) {
133 extension++; // Skip the dot
134 for (size_t i = 0; i < emoji_extension_map_size; i++) {
135 if (strcasecmp(extension, emoji_extension_map[i].key) == 0) {
136 return safe_strdup(emoji_extension_map[i].emoji);
137 }
138 }
139 }
140
141 // Check for hidden files
142 if (filename[0] == '.') {
143 return safe_strdup("⚙️");
144 }
145
146 // Check for executable files
147 if (is_executable(path)) {
148 return safe_strdup("💾");
149 }
150
151 // Check for text files
152 if (is_text_file(path)) {
153 return safe_strdup("📝");
154 }
155
156 // Default emoji for unclassified files
157 return safe_strdup("❓");
158}

161 * @brief Checks if a file is executable
162 *
163 * @param path The path to the file
164 * @return 1 if the file is executable, 0 otherwise
165 */
166int is_executable(const char *path) {
167 struct stat st;
168 if (stat(path, &st) == 0) {
169 return (st.st_mode & S_IXUSR) != 0; // Check if the user execute bit is set
170 }
171 return 0;
172}

174/**
175 * @brief Checks if a file is a text file by examining its contents
176 *
177 * @param path The path to the file
178 * @return 1 if the file is likely a text file, 0 otherwise
179 */
180int is_text_file(const char *path) {
181 FILE *file = fopen(path, "rb");
182 if (!file) {
183 return 0; // Cannot open file, assume it's not text
184 }
185
186 unsigned char buffer[1024];
187 size_t bytesRead = fread(buffer, 1, sizeof(buffer), file);
188 fclose(file);
189
190 if (bytesRead == 0) {
191 return 1; // Empty file is considered text
192 }
193
194 // Check if all read bytes are printable or whitespace
195 for (size_t i = 0; i < bytesRead; i++) {
196 if (!isprint(buffer[i]) && !isspace(buffer[i])) {
197 return 0; // Non-printable, non-whitespace character found
198 }
199 }
200 return 1; // All characters are printable or whitespace
201}
