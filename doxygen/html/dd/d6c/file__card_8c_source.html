21/**
22 * @brief Creates a new file card based on the given path.
23 *
24 * This function initializes a FileCardInfo structure with information about the file,
25 * including its name, emoji representation, and flags for directory and hidden status.
26 *
27 * @param entry Pointer to a FileCardInfo structure to be filled.
28 * @param path Path of the file to analyze.
29 * @return 0 on success, -1 on error.
30 */
31int create_file_entry(FileCardInfo *entry, const char *path)
32{
33 // Duplicate the file name and store it in the entry
34 entry->name = strdup(path);
35 // Get the appropriate emoji for the file
36 entry->emoji = get_emoji(path);
37
38 struct stat path_stat;
39 // Get file status information
40 if (lstat(path, &path_stat) != 0)
41 {
42 return -1; // Return -1 if unable to get file status
43 }
44
45 // Set directory flag based on file type
46 entry->is_directory = S_ISDIR(path_stat.st_mode);
47 // Set hidden flag if the file name starts with a dot
48 entry->is_hidden = (path[0] == '.');
49 // Initialize Git status as empty
50 entry->git_status[0] = '\0';
51 entry->git_status[1] = '\0';
52
53 return 0; // Return 0 on success
54}

70/**
71 * @brief Extracts the file extension from a filename.
72 *
73 * This function finds the last occurrence of a dot in the filename and
74 * returns the substring following it as the file extension.
75 *
76 * @param name The filename to analyze.
77 * @return A pointer to the file extension, or an empty string if no extension is found.
78 */
79char *get_extension(const char *name)
80{
81 char *dot = strrchr(name, '.'); // Find the last occurrence of '.'
82 if (!dot || dot == name)
83 { // If no '.' found or it's the first character
84 return ""; // Return empty string (no extension)
85 }
86 return dot + 1; // Return pointer to character after the last '.'
87}

89/**
90 * @brief Compare two strings case-insensitively.
91 *
92 * This function provides a case-insensitive string comparison,
93 * similar to strcasecmp, but implemented for portability.
94 *
95 * @param a The first string to compare.
96 * @param b The second string to compare.
97 * @return An integer less than, equal to, or greater than zero if a is found,
98 * respectively, to be less than, to match, or be greater than b.
99 */
100static int strcasecmp_wrapper(const char *a, const char *b) {
101 // Loop continues as long as both strings have characters
102 while (*a && *b) {
103 // Convert current characters to lowercase for case-insensitive comparison
104 int ca = tolower((unsigned char)*a);
105 int cb = tolower((unsigned char)*b);
106
107 // If characters differ, return their difference (determines sorting order)
108 if (ca != cb) return ca - cb;
109
110 // Move to the next character in both strings
111 a++;
112 b++;
113 }
114
115 // If we've reached here, one string might be longer than the other
116 // Compare the last characters (one might be '\0' if strings are different lengths)
117 return tolower((unsigned char)*a) - tolower((unsigned char)*b);
118}

120/**
121 * @brief Compare two file entries for sorting.
122 *
123 * This function implements the sorting logic for file entries:
124 * 1. Directories before files
125 * 2. Within directories and files, dotfiles (hidden) before non-dotfiles
126 * 3. Files sorted by extension
127 * 4. Alphabetical sorting within the same type and extension
128 *
129 * @param a Pointer to the first FileCardInfo structure.
130 * @param b Pointer to the second FileCardInfo structure.
131 * @return An integer less than, equal to, or greater than zero if a is found,
132 * respectively, to be less than, to match, or be greater than b.
133 */
134int compare_file_entries(const void *a, const void *b)
135{
136 const FileCardInfo *entry_a = (const FileCardInfo *)a;
137 const FileCardInfo *entry_b = (const FileCardInfo *)b;
138
139 // Sort directories before files
140 if (entry_a->is_directory != entry_b->is_directory) {
141 return entry_b->is_directory - entry_a->is_directory;
142 }
143
144 // For directories, sort dotfiles first
145 if (entry_a->is_directory && entry_b->is_directory) {
146 // Check if one is a dotfile and the other isn't
147 if ((entry_a->name[0] == '.') != (entry_b->name[0] == '.')) {
148 return (entry_b->name[0] == '.') - (entry_a->name[0] == '.');
149 }
150 // Both are either dotfiles or non-dotfiles, sort alphabetically
151 return strcasecmp_wrapper(entry_a->name, entry_b->name);
152 }
153
154 // For files, sort dotfiles first
155 if ((entry_a->name[0] == '.') != (entry_b->name[0] == '.')) {
156 return (entry_b->name[0] == '.') - (entry_a->name[0] == '.');
157 }
158
159 // Sort files by extension
160 const char *ext_a = get_extension(entry_a->name);
161 const char *ext_b = get_extension(entry_b->name);
162 int ext_cmp = strcasecmp_wrapper(ext_a, ext_b);
163
164 // If extensions differ, return their comparison result
165 if (ext_cmp != 0) {
166 return ext_cmp;
167 }
168
169 // If extensions are the same, sort alphabetically by name
170 return strcasecmp_wrapper(entry_a->name, entry_b->name);
171}
